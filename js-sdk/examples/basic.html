Mini Debug SDK - 基础示例

使用说明：

1. 点击右下角的 🐛 按钮打开调试面板

2. 点击下面的测试按钮发送网络请求

3. 在调试面板中查看请求记录和进行拦截操作

// 测试函数 function testFetch() { fetch('https://jsonplaceholder.typicode.com/todos/1') .then(response => response.json()) .then(data => console.log('Fetch GET:', data)) } function testFetchPost() { fetch('https://jsonplaceholder.typicode.com/posts', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ title: 'hello', body: 'world', userId: 1 }) }) .then(response => response.json()) .then(data => console.log('Fetch POST:', data)) } function testXHR() { const xhr = new XMLHttpRequest() xhr.open('GET', 'https://jsonplaceholder.typicode.com/users') xhr.onload = () => console.log('XHR GET:', xhr.responseText) xhr.send() } function testXHRPost() { const xhr = new XMLHttpRequest() xhr.open('POST', 'https://httpbin.org/post') xhr.setRequestHeader('Content-Type', 'application/json') xhr.onload = () => console.log('XHR POST:', xhr.responseText) xhr.send(JSON.stringify({ title: 'XHR POST', body: 'test data', userId: 1 })) }
